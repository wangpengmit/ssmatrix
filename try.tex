\begin{coq_example}
Section ex1.

Variable a : int.

Lemma ex1 : (($2 * a) + (a * $4)) = ($6 * a).
Proof.
  rewrite [(a * $4)]mulrC.
  rewrite -[(($2 * a) + ($4 * a))]mulrDl.
  rewrite -[($2 + $4)]/$6.
  by [].
Qed.

End ex1.
